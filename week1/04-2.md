### STL과 자주 사용하는 필수 문법

#### 상수 레퍼런스

```c++
#include <iostream>

using namespace std;

void modify(int value)
{
	value = 10;
	cout << "function modify()" << endl;
	cout << "주소 : " << &value << endl;
	cout << "값 : " << value << endl;
}

int main()
{
	int value = 5;
	cout << "function main()" << endl;
	cout << "주소 : " << &value << endl;
	cout << "값 : " << value << endl;
	modify(value);
	cout << "function main()" << endl;
	cout << "값 : " << value << endl;

	return 0;
}
```
